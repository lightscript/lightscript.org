function fizzBuzz(n = 100) {
  const _arr = [];
  for (let i = 0; i <= n; i++) {
    if (i % 3 === 0 && i % 5 === 0) {
      _arr.push("fizzbuzz");
    } else if (i % 3 === 0) {
      _arr.push("fizz");
    } else if (i % 5 === 0) {
      _arr.push("buzz");
    } else {
      _arr.push(i);
    }
  }
  return _arr;
}
